# It would be better to change these values to a float instead
# So first I create a function that will do so:
def frac2float(frac):
    """
    Converts a fraction object into a float

    Parameters:
        frac (object): a fraction, whose string looks like 'X/Y'

    Returns:
        float: the decimal value of the fraction
    """
    # make the fraction into a string:
    f = str(frac)
    # split the string around the "/" symbol:
    f = f.split("/")
    # convert each part to floats, then divide:
    f = float(f[0]) / float(f[1])
    # return the value:
    return f

# Now apply the function to the rates column and see that rates are now floats:
df['rate'] = df['rate'].apply(frac2float)
print(df['rate'].loc[0:4])
